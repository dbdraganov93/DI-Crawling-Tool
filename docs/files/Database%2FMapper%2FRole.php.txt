<?php

/**
 * Mapperklasse für die Benutzerrollen in der DB
 *
 * Class Marktjagd_Database_Mapper_Role
 */
class Marktjagd_Database_Mapper_Role extends Marktjagd_Database_Mapper_Abstract
{
    /**
     * Returns the database table class, if no one exists,
     * default will be created.
     *
     * @return  Marktjagd_Database_DbTable_Role
     */
    public function getDbTable()
    {
        return parent::getDbTable();
    }

    /**
     * Saves data to database. If the primary key is set,
     * data will be updated.
     *
     * @param Marktjagd_Database_Entity_Role  $oRole Object data
     * @param bool $bNull Save also null values
     *
     * @return int|bool
     */
    public function save(Marktjagd_Database_Entity_Role $oRole, $bNull = false)
    {
        return parent::_save($oRole, $bNull);
    }

    /**
     * Loads data by primary key(s). By multiple primary
     * keys use an array with the values of the primary key columns.
     *
     * @param mixed $mId Primary key(s) value(s)
     * @param Marktjagd_Database_Entity_Role  $oRole Object for data
     *
     * @return bool True if found, otherwise false
     */
    public function find($mId, Marktjagd_Database_Entity_Role $oRole)
    {
        return parent::_find($mId, $oRole);
    }
}
