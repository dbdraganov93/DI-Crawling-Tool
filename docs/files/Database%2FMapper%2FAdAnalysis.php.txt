<?php

class Marktjagd_Database_Mapper_AdAnalysis extends Marktjagd_Database_Mapper_Abstract
{

    /**
     * Returns the database table class, if no one exists,
     * default will be created.
     *
     * @return  Marktjagd_Database_DbTable_AdAnalysis
     */
    public function getDbTable()
    {
        return parent::getDbTable();
    }

    /**
     * Saves data to database. If the primary key is set,
     * data will be updated.
     *
     * @param Marktjagd_Database_Entity_AdAnalysis $oAdAnalysis Object data
     * @param bool $bNull Save also null values
     * @param bool $bForceInsert Force insert
     * @return void
     */
    public function save(Marktjagd_Database_Entity_AdAnalysis $oAdAnalysis, $bNull = false, $bForceInsert = false)
    {
        parent::_save($oAdAnalysis, $bNull, $bForceInsert);
    }

    /**
     * Loads data by primary key(s). By multiple primary
     * keys use an array with the values of the primary key columns.
     *
     * @param mixed $mId Primary key(s) value(s)
     * @param Marktjagd_Database_Entity_AdAnalysis  $oAdAnalysis Object for data
     *
     * @return bool True if found, otherwise false
     */
    public function find($mId, Marktjagd_Database_Entity_AdAnalysis $oAdAnalysis)
    {
        return parent::_find($mId, $oAdAnalysis);
    }
    
    /**
     * 
     * @param type $idCompany
     * @param type $startDate
     * @param type $endDate
     * @param Marktjagd_Database_Collection_AdAnalysis $oAdAnalysis
     */
    public function findAdsForCompanyByIdAndTimeAndType($idCompany, $startDate, $endDate, $type, Marktjagd_Database_Collection_AdAnalysis $oAdAnalysis)
    {
        $result = $this->getDbTable()->findAdsForCompanyByIdAndTimeAndType($idCompany, $startDate, $endDate, $type);
        
        if (count($result) > 0) {
            $oAdAnalysis->setOptions($result);
            return true;
        }

        return false;
    }
    
}
