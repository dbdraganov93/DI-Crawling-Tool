namespace Marktjagd\ApiClient\Resource\Category;

use Marktjagd\ApiClient\Resource;

/**
 * Category collection resource.
 *
 * @category    resource
 * @package     resource
 * @subpackage  category
 * @author      Robert Freigang <robert.freigang@marktjagd.de>
 * @license     Martktjagd GmbH
 * @link        http://www.marktjagd.de
 */
class CategoryCollectionResource extends Resource\CollectionResource
{

    /**
     * Returns the name of the category collection resource.
     *
     * @return string resource name
     */
    protected static function getName()
    {
        return 'categories';
    }

    /**
     * Sets the resource definition.
     *
     * @return void
     */
    protected function setResourceDefinition()
    {
        $this->hasAttribute(new Resource\ResourceAttributeInteger('hits'));
    }

}
