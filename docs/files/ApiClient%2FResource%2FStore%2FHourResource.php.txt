namespace Marktjagd\ApiClient\Resource\Store;

use Marktjagd\ApiClient\Resource;

/**
 * Hour resource.
 *
 * @category    resource
 * @package     resource
 * @subpackage  store
 * @author      Lutz Petzoldt <lutz.petzoldt@marktjagd.de>
 * @license     Martktjagd GmbH
 * @link        http://www.marktjagd.de
 */
class HourResource extends Resource\Resource
{

    protected static $hasCollection = true;

    /**
     * Sets the resource definition.
     *
     * @return void
     */
    protected function setResourceDefinition()
    {
        $this
            ->hasField(new Resource\ResourceFieldString('day_from'))
            ->hasField(new Resource\ResourceFieldString('day_to'))
            ->hasField(new Resource\ResourceFieldString('time_from'))
            ->hasField(new Resource\ResourceFieldString('time_to'));
    }

}
